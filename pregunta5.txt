SELECT BUSINESS_UNIT
      ,EMPLID
      ,STRM
      ,SESSION_CODE
      ,ACAD_CAREER
      ,ACAD_PROG
      ,CD as CURSOS_DESAPROBADOS,PC as PROMEDIO,NC as NUMERO_CREDITO FROM (select * from (SELECT BUSINESS_UNIT
      ,EMPLID
      ,STRM
      ,SESSION_CODE
      ,ACAD_CAREER
      ,ACAD_PROG
      ,LVF_PAR_RENOV
      ,LVF_VALOR_NUM
    FROM PS_UTP_PAR_REN_PRB_TBL) as Tab
PIVOT(SUM(LVF_VALOR_NUM) FOR LVF_PAR_RENOV IN(CD,PC,NC)) AS PVTAB) TABB;
